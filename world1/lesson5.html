For Loops

# Loop through range
for i in range(5):
    print(i)  # 0, 1, 2, 3, 4

# Loop through range with start/end
for i in range(2, 7):
    print(i)  # 2, 3, 4, 5, 6

# Loop with step
for i in range(0, 10, 2):
    print(i)  # 0, 2, 4, 6, 8

While Loops

var count = 0
while count < 5:
    print(count)
    count += 1

Loop Control

# break - exit loop early
for i in range(10):
    if i == 5:
        break
    print(i)  # 0,1,2,3,4

# continue - skip to next iteration
for i in range(5):
    if i == 2:
        continue
    print(i)  # 0,1,3,4

Looping Through Arrays

var enemies = ["Goblin", "Orc", "Dragon"]

# Loop through items
for enemy in enemies:
    print("Fighting ", enemy)

# Loop with index
for i in range(enemies.size()):
    print(i, ": ", enemies[i])
