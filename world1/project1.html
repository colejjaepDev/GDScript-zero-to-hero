Step 2: The Script

Attach this script to your root Control node:

extends Control

var score: int = 0
var click_value: int = 1
var upgrade_cost: int = 10

@onready var score_label = $CenterContainer/VBoxContainer/ScoreLabel
@onready var click_button = $CenterContainer/VBoxContainer/ClickButton
@onready var value_label = $CenterContainer/VBoxContainer/ValueLabel
@onready var upgrade_button = $CenterContainer/VBoxContainer/UpgradeButton

func _ready():
    click_button.pressed.connect(_on_click_button_pressed)
    upgrade_button.pressed.connect(_on_upgrade_button_pressed)
    update_ui()

func _on_click_button_pressed():
    score += click_value
    update_ui()

func _on_upgrade_button_pressed():
    if score >= upgrade_cost:
        score -= upgrade_cost
        click_value += 1
        upgrade_cost = int(upgrade_cost * 1.5)
        update_ui()

func update_ui():
    score_label.text = "Score: " + str(score)
    value_label.text = "Click Value: " + str(click_value)
    upgrade_button.text = "Upgrade (" + str(upgrade_cost) + ")"
    upgrade_button.disabled = score < upgrade_cost
